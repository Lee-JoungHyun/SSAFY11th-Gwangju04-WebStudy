package day0415;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main_1107_리모컨 {

	private static boolean[] select;
	private static boolean[] selected;
	private static int count;
	private static int[] number;
	private static int[] selectNum;
	private static int answer = Integer.MAX_VALUE;
	private static int N;

	public static void main(String[] args) throws IOException {

		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = null;
		N = Integer.parseInt(br.readLine());
		int M = Integer.parseInt(br.readLine());
		selected = new boolean[10];
		Arrays.fill(selected, true);
		st = new StringTokenizer(br.readLine());
		for (int i = 0; i < M; i++) {
			int num = Integer.parseInt(st.nextToken());
			selected[num] = false;
		}

		// 1234를 자리수로 나누기
		int n = N;
		count = 0;
		int sum = 0;
		number = new int[7];// 1~500000
		selectNum = new int[7];
		while (n / 10 >= 0) {
			int num = n % 10;
			number[count] = num;
			n = n / 10;
			count++;
			if (n == 0) {
				break;
			}

		}

		dfs(0);

		answer = Math.min((Math.abs(N - 100)), answer);

		System.out.println(answer);
	}

	private static void dfs(int cnt) {
		if (cnt == count) {
			// count = 목표 자리수
			int sum = 0;
			for (int i = 0; i < count; i++) {
				if (selectNum[i] != 0)
					sum += (selectNum[i] * Math.pow(10, count));
			}
			answer = Math.min(answer, Math.abs(sum - N) + count);
			return;
		}
		int min = Integer.MAX_VALUE;
		for (int i = 0; i <= 9; i++) {
			// 중복순열
			if (selected[i]) {

				if (min > Math.abs(i - number[cnt])) {
					min = Math.abs(i - number[cnt]);
					selectNum[cnt] = i;
					dfs(cnt + 1);
				}
			}
		}
	}

}
